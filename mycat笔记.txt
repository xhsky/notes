	安装
    0.安装MySQL
    1.安装jdk
    2.安装Mycat
      .# wget -c http://dl.mycat.io/1.6.7.1/Mycat-server-1.6.7.1-release-20190213150257-linux.tar.gz
      .# tar -xf 
      .# vim .bashrc
        export JAVA_HOME=/opt/jdk
        export MYCAT_HOME=/opt/mycat
        export PATH=$JAVA_HOME/bin:$MYCAT_HOME/bin:$PATH
      .# ./bin/mycat start
